CREATE TABLE Students (
    s_id INT PRIMARY KEY,
    name VARCHAR(50),
    dept VARCHAR(50),
    marks INT,
    city VARCHAR(50)
);

-- UPDATE
UPDATE Students
SET name = 'ABHI'
WHERE s_id = 1;

-- DELETE
DELETE FROM Students
WHERE name = 'ABHI';

-- PATTERN MATCHING
SELECT * FROM Students
WHERE name LIKE 'S%';

-- UNION
SELECT city FROM Students
UNION
SELECT dept FROM Students;

-- STRING FUNCTIONS
SELECT LOWER(name) FROM Students;
SELECT LENGTH(name) FROM Students;
